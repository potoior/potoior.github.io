private String executeHttpRequest(ServiceInstance instance, 
                                String endpoint, String requestBody) {
    String url = "http://" + instance.getHost() + ":" + instance.getPort() + endpoint;
    
    try {
        ModernHttpClient client = new ModernHttpClient();
        
        if (requestBody == null) {
            return client.sendGetRequest(url);
        } else {
            return client.sendPostRequest(url, requestBody);
        }
        
    } catch (Exception e) {
        throw new RuntimeException("服务调用失败：" + e.getMessage(), e);
    }
}
